5/*!
6This is a [CoreMIDI](https://developer.apple.com/documentation/coremidi) library for Rust built on top of the low-level bindings [coremidi-sys](https://github.com/jonas-k/coremidi-sys).
7CoreMIDI is a macOS framework that provides APIs for communicating with MIDI (Musical Instrument Digital Interface) devices, including hardware keyboards and synthesizers.
8
9This library preserves the fundamental concepts behind the CoreMIDI framework, while being Rust idiomatic. This means that if you already know CoreMIDI, you will find very easy to start using it.
10
11Please see the [examples](https://github.com/chris-zen/coremidi/tree/master/examples) for getting an idea of how it looks like, but if you are eager to see an example, this is how you would send some note:
12
13```rust,no_run
14use coremidi::{Client, Destination, EventBuffer, Protocol};
15use std::time::Duration;
16use std::thread;
17
18fn main() {
19    let client = coremidi::Client::new("example-client").unwrap();
20    let output_port = client.output_port("example-port").unwrap();
21    let destination = Destination::from_index(0).unwrap();
22    let note_on = EventBuffer::new(Protocol::Midi10).with_packet(0, &[0x2090407f]);
23    let note_off = EventBuffer::new(Protocol::Midi10).with_packet(0, &[0x2080407f]);
24    output_port.send(&destination, &note_on).unwrap();
25    thread::sleep(Duration::from_millis(1000));
26    output_port.send(&destination, &note_off).unwrap();
27}
28```
29
30If you are looking for a portable MIDI library then you can look into:
31
32- [midir](https://github.com/Boddlnagg/midir) (which is using this lib)
33- [portmidi-rs](https://github.com/musitdev/portmidi-rs)
34
35For handling low level MIDI data you may look into:
36
37- [midi-rs](https://github.com/samdoshi/midi-rs)
38- [rimd](https://github.com/RustAudio/rimd)
39
40*/

76/// Unschedules previously-sent packets for all the endpoints.
77/// See [MIDIFlushOutput](https://developer.apple.com/documentation/coremidi/1495312-midiflushoutput).
78///
79pub fn flush() -> Result<(), OSStatus> {
80    let status = unsafe { MIDIFlushOutput(0) };
81    unit_result_from_status(status)
82}
83
84/// Stops and restarts MIDI I/O.
85/// See [MIDIRestart](https://developer.apple.com/documentation/coremidi/1495146-midirestart).
86///
87pub fn restart() -> Result<(), OSStatus> {
88    let status = unsafe { MIDIRestart() };
89    unit_result_from_status(status)
90}
91
92/// Convert an OSStatus into a Result<T, OSStatus> given a mapping closure
93fn result_from_status<T, F: FnOnce() -> T>(status: OSStatus, f: F) -> Result<T, OSStatus> {
94    match status {
95        0 => Ok(f()),
96        _ => Err(status),
97    }
98}
99
100/// Convert an OSStatus into a Result<(), OSStatus>
101fn unit_result_from_status(status: OSStatus) -> Result<(), OSStatus> {
102    result_from_status(status, || ())
