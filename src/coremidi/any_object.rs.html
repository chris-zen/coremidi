1use coremidi_sys::{MIDIObjectRef, MIDIObjectType};
2
3use crate::{Destination, Device, Entity, Object, Source};

5#[derive(Debug, PartialEq, Eq)]
6pub enum AnyObject {
7    Other(Object),
8    Device(Device),
9    Entity(Entity),
10    Source(Source),
11    Destination(Destination),
12    ExternalDevice(Device),
13    ExternalEntity(Entity),
14    ExternalSource(Source),
15    ExternalDestination(Destination),
16}
17
18impl AnyObject {
19    pub(crate) fn create(object_type: MIDIObjectType, object_ref: MIDIObjectRef) -> Option<Self> {
20        match object_type {
21            coremidi_sys::kMIDIObjectType_Other => Some(Self::Other(Object(object_ref))),
22            coremidi_sys::kMIDIObjectType_Device => Some(Self::Device(Device::new(object_ref))),
23            coremidi_sys::kMIDIObjectType_Entity => Some(Self::Entity(Entity::new(object_ref))),
24            coremidi_sys::kMIDIObjectType_Source => Some(Self::Source(Source::new(object_ref))),
25            coremidi_sys::kMIDIObjectType_Destination => {
26                Some(Self::Destination(Destination::new(object_ref)))
27            }
28            coremidi_sys::kMIDIObjectType_ExternalDevice => {
29                Some(Self::ExternalDevice(Device::new(object_ref)))
30            }
31            coremidi_sys::kMIDIObjectType_ExternalEntity => {
32                Some(Self::ExternalEntity(Entity::new(object_ref)))
33            }
34            coremidi_sys::kMIDIObjectType_ExternalSource => {
35                Some(Self::ExternalSource(Source::new(object_ref)))
36            }
37            coremidi_sys::kMIDIObjectType_ExternalDestination => {
38                Some(Self::ExternalDestination(Destination::new(object_ref)))
39            }
40            _ => None,
41        }
42    }
43}
44
45impl AsRef<Object> for AnyObject {
46    fn as_ref(&self) -> &Object {
47        match self {
48            Self::Other(object) => object,
49            Self::Device(device) => &device.object,
50            Self::Entity(entity) => &entity.object,
51            Self::Source(source) => &source.object,
52            Self::Destination(destination) => &destination.object,
53            Self::ExternalDevice(device) => &device.object,
54            Self::ExternalEntity(entity) => &entity.object,
55            Self::ExternalSource(source) => &source.object,
56            Self::ExternalDestination(destination) => &destination.object,
57        }
58    }
59}
